/*
 * A lightning fast primes calculator, uses a varity of optimizations,
 * including storing found primes in an arraylist.
 */

using System;
using System.Collections;

namespace FastPrimes
{
    class Program
    {
        static void Main()
        {
            int currentPrime = 2;
            ArrayList primes = new ArrayList();
            primes.Add(2);
            loop:
            currentPrime++;
            foreach(int test in primes)
                if(currentPrime%test==0)
                    goto loop;
            primes.Add(currentPrime);
            Console.Out.WriteLine(currentPrime);
            goto loop;
        }
    }
}
