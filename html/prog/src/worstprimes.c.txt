/*
 * A small primes printing program.
 * Points to note:
 * Is small - Only four lines.
 * Uses a single variable assignment,
 * one printf statement,
 * and two gotos.
 * Only compiles under GCC.
 * Breaks pretty much every rule about gotos.
 * Actually runs faster than a primes program with loops!
 * Possibly the worst piece of code I have ever written!
 *
 * Have fun debugging!
 */

main(A,B)
{
A:B=A++-~-A;
B:goto*(A==++B)?&&C:(A%B)?&&B:&&D;
C:printf("%i ",A);
D:goto*&&A;
}
