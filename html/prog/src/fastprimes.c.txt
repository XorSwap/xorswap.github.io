/*
 * A C version of my C# fast primes calculator, runs much faster.
 * On my computer, it reaches six digits in under a second.
 */

#define L 999999 //make this number as large as possible, without segfaulting.
main()
{
    int p=0,_[L],i=2,t,m;
    _[p++]=2;
    l:
    i++;
    for(t=0;t<p;t++)
        if(i%_[t]==0)
            goto l;
    _[p++]=i;
    printf("%lld ",i);
    goto l;
}
